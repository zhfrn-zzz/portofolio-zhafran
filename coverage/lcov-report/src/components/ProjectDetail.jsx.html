const handleGithubClick = (githubLink, t) => {
  if (githubLink === 'Private') {
    Swal.fire({
      icon: 'info',
  title: t ? t('project.privateRepo.title', 'Private Source Code') : 'Private Source Code',
  text: t ? t('project.privateRepo.text', 'Sorry, the source code for this project is private.') : 'Sorry, the source code for this project is private.',
  confirmButtonText: t ? t('project.privateRepo.confirm', 'Understood') : 'OK',
      confirmButtonColor: '#3085d6',
      background: '#030014',
      color: '#ffffff'
    });
    return false;
  }
  return true;
};
